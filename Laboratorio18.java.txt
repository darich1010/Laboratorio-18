import java.util.ArrayList;
import java.util.Random;

public class Laboratorio18 {
    public static void main(String[] args) {
        Random rand = new Random();

        
        ArrayList<Soldado> ejercito1 = new ArrayList<>();
        ArrayList<Soldado> ejercito2 = new ArrayList<>();

        
        for (int i = 0; i < 10; i++) {
            ejercito1.add(generarSoldadoAleatorio(i + 1, rand));
            ejercito2.add(generarSoldadoAleatorio(i + 11, rand));
        }

       
        System.out.println("Ejército 1:");
        for (Soldado soldado : ejercito1) {
            soldado.mostrarInfo();
        }

        System.out.println("\nEjército 2:");
        for (Soldado soldado : ejercito2) {
            soldado.mostrarInfo();
        }
    }

    public static Soldado generarSoldadoAleatorio(int id, Random rand) {
        int tipo = rand.nextInt(3);
        int vida = 3 + rand.nextInt(3); 
        int fila = rand.nextInt(10);
        int columna = rand.nextInt(10);

        switch (tipo) {
            case 0:
                return new Espadachin("Espadachin" + id, vida, fila, columna, 5 + rand.nextInt(5));
            case 1:
                return new Arquero("Arquero" + id, vida, fila, columna, 10 + rand.nextInt(5));
            default:
                return new Caballero("Caballero" + id, vida, fila, columna);
        }
    }
}
